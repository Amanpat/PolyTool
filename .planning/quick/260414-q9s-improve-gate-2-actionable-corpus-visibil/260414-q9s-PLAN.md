<objective>
Improve Gate 2 actionable-corpus visibility so the operator can immediately see
(a) which tapes/candidates are actually eligible and why, (b) why most current
tapes are not eligible (especially Silver tapes), and (c) the distribution of
reject reasons and confidence tiers across the corpus.

Purpose: The operator currently has to mentally assemble Gate 2 corpus status
from scattered per-tape fields. The tooling already computes confidence tiers,
reject codes, and diagnostic enrichment, but three gaps prevent clear operator
understanding:
  1. Gate2RankScore drops events_scanned and confidence_class during the
     CandidateResult -> score_gate2_candidate transformation, so the ranked
     table always shows "-" for Events and Conf columns.
  2. tape-manifest prints per-tape rows but no aggregate breakdown of WHY
     the corpus is blocked (reject-code distribution, confidence-tier
     distribution, event-density histogram).
  3. The corpus note mentions "executable_ticks > 0" but never explains that
     Silver tapes are structurally unusable because they contain only
     price_2min_guide events with no L2 book data.

Output: Updated scorer.py, scan_gate2_candidates.py, tape_manifest.py,
expanded test file, and mandatory dev log.
</objective>

class TestCorpusQualityBreakdown:
    def _capture_breakdown(self, records, summary):
        """Capture stdout from print_corpus_quality_breakdown."""
        from tools.cli.tape_manifest import print_corpus_quality_breakdown
        buf = io.StringIO()
        old_stdout = sys.stdout
        sys.stdout = buf
        try:
            print_corpus_quality_breakdown(records, summary)
        finally:
            sys.stdout = old_stdout
        return buf.getvalue()

    def _make_summary(self, eligible=0, total=5):
        from tools.cli.tape_manifest import CorpusSummary
        return CorpusSummary(
            total_tapes=total,
            eligible_count=eligible,
            ineligible_count=total - eligible,
            by_regime={"politics": {"total": 2, "eligible": 0}, "sports": {"total": 2, "eligible": 0}, "new_market": {"total": 1, "eligible": 0}, "unknown": {"total": 0, "eligible": 0}},
            mixed_regime_eligible=False,
            gate2_eligible_tapes=[],
            generated_at="2026-04-14T00:00:00+00:00",
            corpus_note="BLOCKED",
            regime_coverage={"covered_regimes": [], "missing_regimes": ["politics", "sports", "new_market"]},
        )

    def test_reject_code_distribution_printed(self):
        records = [
            _make_record(evidence={"ticks_with_depth_ok": 5, "ticks_with_edge_ok": 0, "ticks_with_depth_and_edge": 0}),
            _make_record(evidence={"ticks_with_depth_ok": 0, "ticks_with_edge_ok": 3, "ticks_with_depth_and_edge": 0}),
            _make_record(reject_reason="no events.jsonl found"),
        ]
        # Ensure diagnostics are populated
        for r in records:
            if not r.diagnostics:
                from tools.cli.tape_manifest import enrich_tape_diagnostics
                r.diagnostics = enrich_tape_diagnostics(r)
        summary = self._make_summary(total=3)
        output = self._capture_breakdown(records, summary)
        assert "DEPTH_ONLY" in output
        assert "EDGE_ONLY" in output
        assert "NO_EVENTS" in output

    def test_confidence_tier_distribution_printed(self):
        records = [
            _make_record(recorded_by="watch-arb-candidates", evidence={"events_scanned": 100, "ticks_with_both_bbo": 50}),
            _make_record(recorded_by="prepare-gate2", evidence={"events_scanned": 30, "ticks_with_both_bbo": 0}),
        ]
        for r in records:
            if not r.diagnostics:
                from tools.cli.tape_manifest import enrich_tape_diagnostics
                r.diagnostics = enrich_tape_diagnostics(r)
        summary = self._make_summary(total=2)
        output = self._capture_breakdown(records, summary)
        assert "GOLD" in output
        assert "SILV" in output or "SILVER" in output

    def test_silver_warning_when_blocked(self):
        records = [
            _make_record(recorded_by="prepare-gate2", evidence={"events_scanned": 30, "ticks_with_both_bbo": 0}),
        ]
        for r in records:
            r.diagnostics = enrich_tape_diagnostics(r)
        summary = self._make_summary(eligible=0, total=1)
        output = self._capture_breakdown(records, summary)
        assert "Silver" in output or "SILVER" in output or "structurally unusable" in output

    def test_no_silver_warning_when_eligible_exists(self):
        records = [
            _make_record(recorded_by="watch-arb-candidates", eligible=True, evidence={"events_scanned": 100, "ticks_with_both_bbo": 50, "ticks_with_depth_and_edge": 5, "ticks_with_depth_ok": 10, "ticks_with_edge_ok": 8}),
        ]
        for r in records:
            r.diagnostics = enrich_tape_diagnostics(r)
        summary = self._make_summary(eligible=1, total=1)
        output = self._capture_breakdown(records, summary)
        # Should NOT show the Silver warning when eligible tapes exist
        assert "structurally unusable" not in output

    def test_next_action_capture_gold(self):

        summary = self._make_summary(eligible=0, total=1)
        output = self._capture_breakdown(records, summary)
